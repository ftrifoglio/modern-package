[docs]def add(*nums: int) -> int:
    """
    Add all numbers.

    Parameters
    ----------
    nums : int
        Numbers to add. Comma separated.

    Returns
    -------
    int
        Sum of all numbers.

    """
    return reduce(lambda a, b: a + b, list(nums))
